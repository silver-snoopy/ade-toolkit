# ADE CrewAI Runner Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the CrewAI runner module (`python -m ade.crew run`) that executes local LLM agents for coding, testing, and fixing within git worktrees.

**Architecture:** A Python subprocess dispatched by Claude Code. It loads agent definitions from `.ade/crew/*.yaml`, creates CrewAI agents with sandboxed tools (SafeShellTool, SafeFileTool), runs the appropriate phase (stubs/code/test/fix), reports progress to a log file, and exits with structured codes (0-3).

**Tech Stack:** Python 3.11+, CrewAI (agent framework), Ollama (LLM runtime), PyYAML (config loading)

**Reference:** Full architecture spec at `docs/ade-architecture-design.md`, sections: CrewAI Agent Definitions, Security Model, CrewAI Subprocess Lifecycle Management, Observability.

**Prerequisite:** Plan 1 (Core CLI) must be merged first — this module reads `.ade/config.yaml` generated by `ade init`.

```python
"""SafeShellTool — command execution with allowlist enforcement."""

from __future__ import annotations

import shlex
import subprocess
from pathlib import Path

from crewai.tools import BaseTool
from pydantic import Field


# Commands that agents are allowed to run (matched by first token)
_ALLOWED_PREFIXES: list[str] = [
    # Test runners
    "pytest", "npx jest", "npm test", "go test", "cargo test",
    # Linters / scanners
    "ruff", "eslint", "prettier", "semgrep",
    # Pre-commit
    "pre-commit",
    # Git (read + stage + commit only)
    "git add", "git commit", "git diff", "git status", "git log", "git show",
    # Build tools
    "npm run", "make", "cargo build",
]

# Commands that are explicitly blocked (even if they match an allowed prefix)
_BLOCKED_PATTERNS: list[str] = [
    "rm -rf", "rm -r",
    "curl", "wget", "ssh", "scp",
    "docker", "sudo",
    "pip install", "npm install",
    "git push", "git checkout main", "git checkout master",
    "git reset --hard", "git clean",
]


def is_command_allowed(command: str) -> bool:
    """Check if a command is allowed by the sandbox policy."""
    cmd = command.strip()

    # Check blocklist first (takes priority)
    for pattern in _BLOCKED_PATTERNS:
        if cmd.startswith(pattern) or f" {pattern}" in cmd:
            return False

    # Check allowlist
    for prefix in _ALLOWED_PREFIXES:
        if cmd.startswith(prefix):
            return True

    return False


class SafeShellTool(BaseTool):
    """Execute shell commands within a sandboxed worktree directory."""

    name: str = "execute_command"
    description: str = (
        "Execute a shell command in the project worktree. "
        "Only allowlisted commands (test runners, linters, git) are permitted. "
        "Network access, destructive operations, and dependency changes are blocked."
    )
    worktree_path: Path = Field(description="Path to the git worktree")
    timeout: int = Field(default=300, description="Command timeout in seconds")

    def _run(self, command: str) -> str:
        if not is_command_allowed(command):
            return f"BLOCKED: '{command}' is not in the allowed command list"

        try:
            result = subprocess.run(
                command,
                shell=True,
                cwd=self.worktree_path,
                capture_output=True,
                text=True,
                timeout=self.timeout,
            )
            output = result.stdout
            if result.returncode != 0:
                output += f"\n[exit code: {result.returncode}]\n{result.stderr}"
            return output or "(no output)"
        except subprocess.TimeoutExpired:
            return f"TIMEOUT: Command '{command}' timed out after {self.timeout}s"
        except Exception as e:
            return f"ERROR: {e}"
```

import sys
import click

from ade.crew.runner import run


@click.group()
def cli():
    pass


@cli.command()
@click.option("--phase", required=True, type=click.Choice(["stubs", "code", "test", "fix"]))
@click.option("--task-id", required=True)
@click.option("--worktree", required=True)
